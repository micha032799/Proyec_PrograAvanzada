CREATE PROCEDURE RegistrarCuentaSP
	@Identificacion varchar(25),
	@Nombre			varchar(100),
	@Correo			varchar(100),
	@Contrasena		varchar(25),
	@Estado			bit,
	@Direccion		varchar(500)
AS
BEGIN

	INSERT INTO dbo.TUsuario (Identificacion, Nombre, Correo, Contrasena, Estado, Direccion)
	VALUES (@Identificacion, @Nombre, @Correo, @Contrasena, @Estado, @Direccion)

END
GO


CREATE PROCEDURE IniciarSesionSP
	@Correo			varchar(100),
	@Contrasena		varchar(25)
AS
BEGIN

	SELECT ConUsuario,
		   Identificacion,
		   Nombre,
		   Correo,
		   Contrasena,
		   Estado,
		   Direccion
	FROM dbo.TUsuario
	WHERE  Correo = @Correo
	AND	   Contrasena = @Contrasena
	AND	   Estado = 1

END
GO
